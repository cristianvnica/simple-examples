HTML Page example

HTML page structure


<!DOCTYPE html>
<html lang="en">
  <head>
    <title>Page title</title>
  </head>
    
  <body>
    <header>
      <h1>page title</h1>
    </header>
    <main>
      <p>main content</p>
    </main>
    <aside>
      <p>side content</p>
    </aside>
    <footer>
      <p>footer</p>
    </footer>
  </body>
</html>
